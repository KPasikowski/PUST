function [ y, u, E, yzad ] = policzPID( Kp_, Ti_, Td_, Kk_)
U_min = -1;
U_max = 1;

Kp = Kp_;
Ti = Ti_;
Td = Td_;

Kk = Kk_;
Tp = 0.5;

r2 = (Kp * Td) / Tp ;
r1 = Kp * ( (Tp/(2*Ti)) - 2*(Td/Tp) - 1 ) ;
r0 = Kp * ( 1 + Tp/(2*Ti) + Td/Tp ) ;

%warunki poczatkowe
